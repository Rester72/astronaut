using System.Collections;
using UnityEngine;
using UnityEngine.UI;
using TMPro; // ✅ Required

public class DialogueManager : MonoBehaviour
{
    [Header("UI")]
    public GameObject chatBox;
    public TextMeshProUGUI dialogueText; // ✅ FIXED
    public Image portrait;

    [Header("Typing Settings")]
    public float typingSpeed = 0.03f;

    private DialogueLine[] lines;
    private int index;

    private bool isTyping = false;
    private bool canContinue = false;

    void Start()
    {
        chatBox.SetActive(false);
    }

    void Update()
    {
        if (chatBox.activeSelf && Input.GetMouseButtonDown(0))
        {
            // If still typing → complete instantly
            if (isTyping)
            {
                StopAllCoroutines();
                dialogueText.text = lines[index].text;
                portrait.sprite = lines[index].speakerImage;

                isTyping = false;
                canContinue = true;
            }
            // If finished → next line
            else if (canContinue)
            {
                NextLine();
            }
        }
    }

    public void StartDialogue(DialogueLine[] dialogueLines)
    {
        lines = dialogueLines;
        index = 0;

        chatBox.SetActive(true);
        StartCoroutine(TypeLine());
    }

    IEnumerator TypeLine()
    {
        isTyping = true;
        canContinue = false;

        dialogueText.text = "";
        portrait.sprite = lines[index].speakerImage;

        foreach (char c in lines[index].text.ToCharArray())
        {
            dialogueText.text += c;
            yield return new WaitForSeconds(typingSpeed);
        }

        isTyping = false;
        canContinue = true;
    }

    void NextLine()
    {
        index++;

        if (index < lines.Length)
        {
            StartCoroutine(TypeLine());
        }
        else
        {
            chatBox.SetActive(false);
        }
    }
}
